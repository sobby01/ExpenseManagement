CREATE PROCEDURE "Sel_Stock"
RETURNS(
  PRODUCTBRAND VARCHAR(50) CHARACTER SET ISO8859_1 COLLATE ISO8859_1,
  PRODUCTDISCOUNT SMALLINT,
  PRODUCTID INTEGER,
  PRODUCTNAME VARCHAR(50) CHARACTER SET ISO8859_1 COLLATE ISO8859_1,
  PRODUCTPRICE INTEGER)
AS
BEGIN
FOR
SELECT	
STOCK.STOCK_PRODUCTBRAND,
STOCK.STOCK_PRODUCTDISCOUNT,
STOCK.STOCK_PRODUCTID,
STOCK.STOCK_PRODUCTNAME,
STOCK.STOCK_PRODUCTPRICE
FROM	STOCK
INTO
        :PRODUCTBRAND ,
		:PRODUCTDISCOUNT ,
		:PRODUCTID ,
		:PRODUCTNAME ,
		:PRODUCTPRICE
DO
SUSPEND;
END;
